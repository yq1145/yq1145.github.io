二: 多任务异步协程: ​

1. 导入包 ​

python
import asyncio
import aiohttp
import random
from concurrent.futures import ThreadPoolExecutor
import aiofiles
import requests

2.创建异步方法 ​

python
async def get_image(dict):
    image_url = dict['url']
    image_name = dict['name']
    print(image_name, "正在下载.....")
    async with aiohttp.ClientSession() as session:
        async with session.get(image_url, headers=random.choice(headers)) as response:
            async with aiofiles.open("./img/" + image_name, mode="wb") as f:
                await f.write(await response.read())
                print(image_name, "下载完成.....")

使用异步的aiofiles写入数据

3.创建任务列表,并将任务放入任务列表中 ​

python
tasks = []
  # 创建任务
        task = asyncio.ensure_future(get_image(dict=dict))
        # 将任务放入任务列表中
        tasks.append(task)

4.开启循环,将任务挂起执行 ​

python
# 开启循环
loop = asyncio.get_event_loop()
# 将任务挂起
loop.run_until_complete(asyncio.wait(tasks))

# -*- coding = utf-8 -*-
# @Time :  22:00 
# @Author :  yangqi
# @File : _多任务异步协程下载图片.py
# @software: PyCharm
import asyncio
import aiohttp
import random
from concurrent.futures import ThreadPoolExecutor
import aiofiles
import requests

headers = [
    {
        'User-Agent': "Mozilla/5.0 (X11; U; Linux x86_64; zh-CN; rv:1.9.2.10) Gecko/20100922 Ubuntu/10.10 (maverick) Firefox/3.6.10"},
    {
        'User-Agent': "Mozilla/5.0 (Windows NT 6.1; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/39.0.2171.95 Safari/537.36 OPR/26.0.1656.60"},
    {
        'User-Agent': "Mozilla/5.0 (compatible; MSIE 9.0; Windows NT 6.1; WOW64; Trident/5.0; SLCC2; .NET CLR 2.0.50727; .NET CLR 3.5.30729; .NET CLR 3.0.30729; Media Center PC 6.0; .NET4.0C; .NET4.0E; QQBrowser/7.0.3698.400)"},
    {
        'User-Agent': "Mozilla/4.0 (compatible; MSIE 7.0; Windows NT 5.1; Trident/4.0; SV1; QQDownload 732; .NET4.0C; .NET4.0E; SE 2.X MetaSr 1.0)"},
    {'User-Agent': "Mozilla/5.0 (Windows NT 6.1; WOW64; Trident/7.0; rv:11.0) like Gecko"},
    {
        'User-Agent': "Mozilla/5.0 (Windows NT 6.1; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Maxthon/4.4.3.4000 Chrome/30.0.1599.101 Safari/537.36"}
]
"""
    多任务异步协程:
        1. 创建异步方法
        2. 通过asyncio.ensure_future 创建任务
        3. 将任务放入任务列表中
        4. 开启无限循环,直到任务结束
"""


async def get_image(dict):
    image_url = dict['url']
    image_name = dict['name']
    print(image_name, "正在下载.....")
    async with aiohttp.ClientSession() as session:
        async with session.get(image_url, headers=random.choice(headers)) as response:
            async with aiofiles.open("./img/" + image_name, mode="wb") as f:
                await f.write(await response.read())
                print(image_name, "下载完成.....")


tasks = []
for i in range(1, 10):
    url = f"https://picsum.photos/v2/list?page={i}"
    resp_json = requests.get(url, headers=random.choice(headers)).json()
    for image in resp_json:
        print(image)
        dict = {
            "name": image['id'] + ".jpg",
            "url": image['download_url']
        }
        # 创建任务
        task = asyncio.ensure_future(get_image(dict=dict))
        # 将任务放入任务列表中
        tasks.append(task)
# 开启循环
loop = asyncio.get_event_loop()
# 将任务挂起
loop.run_until_complete(asyncio.wait(tasks))
