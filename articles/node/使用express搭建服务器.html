使用express搭建服务器 ​

一 安装依赖 ​

js
npm install express --save

二 启动 ​

js
node index.js

index.js ​

js

// app.js/index.js
const express = require('express')
const app = express()
const port = 3000

app.get('/', (req, res) => {
  res.send('Hello World!')
})

app.listen(port, () => {
  console.log(`Example app listening on port ${port}`)
})
